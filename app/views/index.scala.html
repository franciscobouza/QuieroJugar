/login

Tipo: GET

Parámetros: user: String, password: String

Retorno: JSON conteniendo retorno: String (OK | ERROR), resultado_login: boolean, id_usuario: int (con retorno OK) y mensaje: String (con retorno ERROR)

/registrar

Tipo: POST

Parámetros: user: String, password: String, nombre: String

Retorno: JSON conteniendo retorno: String (OK | ERROR), idUsuario: int (con retorno OK), mensaje: String (con retorno ERROR)

/getCanchas

Tipo: GET

Retorno: JSON conteniendo retorno: String (OK | ERROR), canchas: [{nombre: String, direccion: String, telefono: String, ubicacion: {latitud: double, longitud: double}, foto_inicial: String}] (con retorno OK), mensaje: String (con retorno ERROR)

/getCancha

Tipo: GET

Parámetros: nombre: String

Retorno: JSON conteniendo retorno: String (OK | ERROR), cancha: {nombre: String, direccion: String, telefono: String, ubicacion: {latitud: double, longitud: double}, foto_inicial: String, fotos: [String]} (con retorno OK), mensaje: String (con retorno ERROR)

/getPartidos

Tipo: GET

Parámetros: nombreCancha: String

Retorno: JSON conteniendo retorno: String (OK | ERROR), partidos: [{id: int, nombre: String, cancha: String, cantidad_jugadores: int}] (con retorno OK), mensaje: String (con retorno ERROR)

/getPartido

Tipo: GET

Parámetros: idPartido: int

Retorno: JSON conteniendo retorno: String (OK | ERROR), partido: {id: int, nombre: String, cancha: String, cantidad_jugadores: int, jugadores: [String]} (con retorno OK), mensaje: String (con retorno ERROR)

/crearPartido

Tipo: POST

Parámetros: nombreCancha: String, nombrePartido: String, idUsuario: int

Retorno: JSON conteniendo retorno: String, idPartido: int (OK | ERROR), mensaje: String (con retorno ERROR)

/inscribirseAPartido

Tipo: POST

Parámetros: idUsuario: int, idPartido: int

Retorno: JSON conteniendo retorno: String (OK | ERROR), mensaje: String (con retorno ERROR)

/noSeSiSeraDemasiado [Requerimiento extra opcional: Obtención de canchas ordenadas por cercanía a partir de ubicación]

Tipo: GET

Parámetros: latO: double, longO: double

Retorno: JSON conteniendo retorno: String (OK | ERROR), canchas: [{nombre: String, direccion: String, telefono: String, ubicacion: {latitud: double, longitud: double}, foto_inicial: String, distancia: double}] (con retorno OK), mensaje: String (con retorno ERROR)
